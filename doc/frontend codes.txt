<script>
  // ‚úÖ Update cart count on load
  function updateCartNotification() {
    const cart = JSON.parse(localStorage.getItem("cart")) || [];
    const cartIcon = document.getElementById("cartCount");
    if (cartIcon) {
      cartIcon.textContent = cart.length;
    }
  }

  // ‚úÖ Add item to cart and update badge
  function addToCart(name, price) {
    let cart = JSON.parse(localStorage.getItem("cart")) || [];
    cart.push({ name: name, price: price });
    localStorage.setItem("cart", JSON.stringify(cart));
    updateCartNotification();
    alert(`${name} added to cart!`);
  }

  // ‚úÖ Category Filter
  function filterByCategory() {
    const selected = document.getElementById("categorySelect").value.toLowerCase();
    const cards = document.querySelectorAll(".product-card");
    cards.forEach(card => {
      const category = card.getAttribute("data-category");
      card.style.display = selected === "all" || category === selected ? "block" : "none";
    });
  }

  // ‚úÖ Search Filter
  function filterProducts() {
    const query = document.getElementById("searchInput").value.toLowerCase();
    const cards = document.querySelectorAll(".product-card");
    cards.forEach(card => {
      const text = card.innerText.toLowerCase();
      card.style.display = text.includes(query) ? "block" : "none";
    });
  }

  // ‚úÖ Call once on page load
  updateCartNotification();
</script>

function updateCartDisplay() {
  cartItemsContainer.innerHTML = "";
  let total = 0;

  cart.forEach((item, index) => {
    total += item.price;

    const itemDiv = document.createElement("div");
    itemDiv.classList.add("cart-item");
    itemDiv.innerHTML = `
      <p><strong>${item.name}</strong> - ‚Çπ${item.price}</p>

    `;
    cartItemsContainer.appendChild(itemDiv);
  });

  const discount = cart.length > 0 ? 0.1 * total : 0; // 10% discount
  const fee = 20;
  const final = total - discount + fee;

  totalPriceElement.textContent = total.toFixed(2);
  discountElement.textContent = discount.toFixed(2);
  finalAmountElement.textContent = final.toFixed(2);

  updateCartNotification();
}

function removeFromCart(index) {
  cart.splice(index, 1);
  localStorage.setItem("cart", JSON.stringify(cart));
  updateCartDisplay();
}

function updateCartNotification() {
  const cartIcon = document.getElementById("cartCount");
  if (cartIcon) {
    cartIcon.textContent = cart.length;
  }
}
